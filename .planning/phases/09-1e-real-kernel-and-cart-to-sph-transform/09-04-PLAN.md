    **Add a new function `collect_1e_sph_matrix_vendor`** that mirrors the existing `collect_1e_sph_matrix` but calls vendored libcint FFI instead of cintx eval_raw:

    ```rust
    #[cfg(has_vendor_libcint)]
    fn collect_1e_sph_matrix_vendor(
        operator: &str,  // "ovlp", "kin", or "nuc"
        atm: &[i32],
        bas: &[i32],
        env: &[f64],
    ) -> Vec<f64> {
        use cintx_oracle::vendor_ffi;

        let natm = (atm.len() / ATM_SLOTS) as i32;
        let nbas = (bas.len() / BAS_SLOTS) as i32;

        // Determine AO count per shell from angular momenta
        let ang: Vec<i32> = (0..N_SHELLS)
            .map(|s| bas[s * BAS_SLOTS + ANG_OF])
            .collect();
        let shell_nao: Vec<usize> = ang.iter().map(|&l| nsph(l)).collect();
        let n_ao: usize = shell_nao.iter().sum();

        let mut matrix = vec![0.0_f64; n_ao * n_ao];

        let mut row_offset = 0usize;
        for si in 0..N_SHELLS {
            let ni = shell_nao[si];
            let mut col_offset = 0usize;
            for sj in 0..N_SHELLS {
                let nj = shell_nao[sj];
                let shls = [si as i32, sj as i32];
                let n_elem = ni * nj;
                let mut out = vec![0.0_f64; n_elem];

                let _ret = match operator {
                    "ovlp" => vendor_ffi::vendor_int1e_ovlp_sph(
                        &mut out, &shls, atm, natm, bas, nbas, env,
                    ),
                    "kin" => vendor_ffi::vendor_int1e_kin_sph(
                        &mut out, &shls, atm, natm, bas, nbas, env,
                    ),
                    "nuc" => vendor_ffi::vendor_int1e_nuc_sph(
                        &mut out, &shls, atm, natm, bas, nbas, env,
                    ),
                    _ => panic!("unknown operator: {operator}"),
                };

                // libcint 1e output is column-major (Fortran order): out[j*ni + i]
                // Convert to row-major for our matrix layout
                for ii in 0..ni {
                    for jj in 0..nj {
                        matrix[(row_offset + ii) * n_ao + (col_offset + jj)] = out[jj * ni + ii];
                    }
                }

                col_offset += nj;
            }
            row_offset += ni;
        }

        matrix
    }
    ```

    **IMPORTANT NOTE on output layout:** Vendored libcint uses column-major (Fortran) output ordering for 1e integrals: the flat buffer is `out[j*ni + i]` where i is bra index and j is ket index. The existing `collect_1e_sph_matrix` function uses `out[ii * nj + jj]` which is the cintx convention (row-major). The vendor function MUST use `out[jj * ni + ii]` to correctly unpack libcint's column-major output. If the tests show systematic mismatches, the first thing to check is this transpose — the executor may need to verify whether it already does the transpose internally.
