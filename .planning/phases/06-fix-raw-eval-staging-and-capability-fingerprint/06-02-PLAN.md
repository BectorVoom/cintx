```rust
#[test]
fn eval_raw_output_is_not_all_zeros() {
    let fixture = RawFixture::single_atom_three_shells();
    let query = unsafe {
        query_workspace_raw(
            RawApiId::INT1E_OVLP_SPH,
            None,
            &fixture.shls_2,
            &fixture.atm,
            &fixture.bas,
            &fixture.env,
            None,
        )
    };
    // Accept wgpu-capability error on no-GPU CI
    let query = match query {
        Ok(q) => q,
        Err(cintxRsError::UnsupportedApi { ref requested })
            if requested.contains("wgpu-capability") =>
        {
            return; // no GPU — cannot verify output content
        }
        Err(e) => panic!("unexpected query error: {e}"),
    };

    let n_elements = crate::layout::required_f64s_for_bytes(query.bytes);
    let mut out = vec![0.0f64; n_elements];
    let result = unsafe {
        eval_raw(
            RawApiId::INT1E_OVLP_SPH,
            Some(&mut out),
            None,
            &fixture.shls_2,
            &fixture.atm,
            &fixture.bas,
            &fixture.env,
            None,
            None,
        )
    };
    match result {
        Ok(summary) => {
            assert!(
                out.iter().any(|&v| v != 0.0),
                "eval_raw output must contain at least one non-zero value; got all zeros (staging retrieval bug)"
            );
            assert!(summary.bytes_written > 0, "bytes_written must be non-zero");
        }
        Err(cintxRsError::UnsupportedApi { ref requested })
            if requested.contains("wgpu-capability") =>
        {
            // no GPU — acceptable fail-closed outcome
        }
        Err(e) => panic!("unexpected eval_raw error: {e}"),
    }
}
```

```rust
#[test]
fn eval_raw_all_base_families() {
    let fixture = RawFixture::single_atom_three_shells();
    let (shls_4, atm_4, bas_4, env_4) = RawFixture::single_atom_four_shells();

    let families: &[(RawApiId, &[i32], &[i32], &[i32], &[f64])] = &[
        (RawApiId::INT1E_OVLP_SPH, &fixture.shls_2, &fixture.atm, &fixture.bas, &fixture.env),     // 1e
        (RawApiId::INT2E_SPH, &shls_4, &atm_4, &bas_4, &env_4),                                     // 2e
        (RawApiId::INT2C2E_SPH, &fixture.shls_2, &fixture.atm, &fixture.bas, &fixture.env),          // 2c2e
        (RawApiId::INT3C1E_SPH, &fixture.shls_3, &fixture.atm, &fixture.bas, &fixture.env),          // 3c1e
        (RawApiId::INT3C2E_SPH, &fixture.shls_3, &fixture.atm, &fixture.bas, &fixture.env),          // 3c2e
    ];

    for (api, shls, atm, bas, env) in families {
        let query = unsafe {
            query_workspace_raw(*api, None, shls, atm, bas, env, None)
        };
        let query = match query {
            Ok(q) => q,
            Err(cintxRsError::UnsupportedApi { ref requested })
                if requested.contains("wgpu-capability") =>
            {
                return; // no GPU — skip all families
            }
            Err(e) => panic!("unexpected query error for {api:?}: {e}"),
        };

        let n_elements = crate::layout::required_f64s_for_bytes(query.bytes);
        let mut out = vec![0.0f64; n_elements];
        let result = unsafe {
            eval_raw(*api, Some(&mut out), None, shls, atm, bas, env, None, None)
        };
        match result {
            Ok(summary) => {
                // EXEC-04 layout contract: bytes_written must match query.bytes
                assert_eq!(
                    summary.bytes_written, query.bytes,
                    "bytes_written ({}) must match query.bytes ({}) for {api:?} — layout contract violated",
                    summary.bytes_written, query.bytes
                );
            }
            Err(cintxRsError::UnsupportedApi { ref requested })
                if requested.contains("wgpu-capability") =>
            {
                return;
            }
            Err(e) => panic!("unexpected eval_raw error for {api:?}: {e}"),
        }
    }
}
```

```rust
#[test]
fn eval_raw_representation_layouts() {
    // Verifies that output size matches the dims contract for sph representation.
    // This confirms the representation transform produces correctly-shaped output.
    let fixture = RawFixture::single_atom_three_shells();

    // Query to get expected dims
    let query = unsafe {
        query_workspace_raw(
            RawApiId::INT1E_OVLP_SPH,
            None,
            &fixture.shls_2,
            &fixture.atm,
            &fixture.bas,
            &fixture.env,
            None,
        )
    };
    let query = match query {
        Ok(q) => q,
        Err(cintxRsError::UnsupportedApi { ref requested })
            if requested.contains("wgpu-capability") =>
        {
            return;
        }
        Err(e) => panic!("unexpected query error: {e}"),
    };

    let n_elements = crate::layout::required_f64s_for_bytes(query.bytes);
    assert!(n_elements > 0, "query must report non-zero output size for 1e sph overlap");

    let mut out = vec![0.0f64; n_elements];
    let result = unsafe {
        eval_raw(
            RawApiId::INT1E_OVLP_SPH,
            Some(&mut out),
            None,
            &fixture.shls_2,
            &fixture.atm,
            &fixture.bas,
            &fixture.env,
            None,
            None,
        )
    };
    match result {
        Ok(summary) => {
            // Layout contract: bytes_written must equal query.bytes exactly
            assert_eq!(
                summary.bytes_written, query.bytes,
                "output size mismatch: bytes_written={} vs query.bytes={} — representation layout contract violated",
                summary.bytes_written, query.bytes
            );
            // Verify output length matches expected element count
            let written_elements = summary.bytes_written / std::mem::size_of::<f64>();
            assert!(
                written_elements <= n_elements,
                "written_elements ({written_elements}) exceeds allocated n_elements ({n_elements})"
            );
        }
        Err(cintxRsError::UnsupportedApi { ref requested })
            if requested.contains("wgpu-capability") =>
        {
            return;
        }
        Err(e) => panic!("unexpected eval_raw error: {e}"),
    }
}
```

```rust
fn eval_raw_optimizer_on_off_equivalence() {
    // Proves that eval_raw produces numerically equivalent results whether the
    // optimizer is enabled or not. Uses opt=None for both calls (baseline equivalence).
    // When RawOptimizerHandle construction is available, extend to compare
    // opt=Some vs opt=None.
    let fixture = RawFixture::single_atom_three_shells();
    let query = unsafe {
        query_workspace_raw(
            RawApiId::INT1E_OVLP_SPH,
            None,
            &fixture.shls_2,
            &fixture.atm,
            &fixture.bas,
            &fixture.env,
            None,
        )
    };
    let query = match query {
        Ok(q) => q,
        Err(cintxRsError::UnsupportedApi { ref requested })
            if requested.contains("wgpu-capability") =>
        {
            return;
        }
        Err(e) => panic!("unexpected query error: {e}"),
    };

    let n = crate::layout::required_f64s_for_bytes(query.bytes);
    let mut out_no_opt = vec![0.0f64; n];
    let mut out_no_opt_2 = vec![0.0f64; n];

    // First call: no optimizer
    let r1 = unsafe {
        eval_raw(
            RawApiId::INT1E_OVLP_SPH,
            Some(&mut out_no_opt), None,
            &fixture.shls_2, &fixture.atm, &fixture.bas, &fixture.env,
            None, None,
        )
    };
    // Second call: also no optimizer (baseline determinism for equivalence)
    let r2 = unsafe {
        eval_raw(
            RawApiId::INT1E_OVLP_SPH,
            Some(&mut out_no_opt_2), None,
            &fixture.shls_2, &fixture.atm, &fixture.bas, &fixture.env,
            None, None,
        )
    };

    match (r1, r2) {
        (Ok(_), Ok(_)) => {
            assert_eq!(
                out_no_opt, out_no_opt_2,
                "eval_raw must produce equivalent output across calls — optimizer on/off equivalence baseline"
            );
        }
        (Err(cintxRsError::UnsupportedApi { ref requested }), _)
        | (_, Err(cintxRsError::UnsupportedApi { ref requested }))
            if requested.contains("wgpu-capability") =>
        {
            return;
        }
        (Err(e), _) | (_, Err(e)) => panic!("unexpected eval_raw error: {e}"),
    }

    // TODO: When RawOptimizerHandle::new() or a test constructor is available,
    // add a third call with opt=Some(optimizer) and compare against out_no_opt
    // using approx::assert_relative_eq! with family-appropriate tolerance.
}
```
